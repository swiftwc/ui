const rand = (n: number) => `${Math.floor(Math.random() * n)}`

function onClick(event: PointerEvent) {
  const el = (event.target as HTMLElement).closest<HTMLElement>('button,summary')
  if (!el) return

  el.setPointerCapture(event.pointerId)

  el.classList.add('active')

  const pointerId = rand(1000)

  event.currentTarget.dataset.lastPointerCapture = pointerId
  el.dataset.pointerCapture = pointerId

  // el.setPointerCapture(event.pointerId)

  el.addEventListener('pointerout', (e) => {
    console.log(99)
    e.stopImmediatePropagation()
    e.stopPropagation()
    e.preventDefault()
    el.releasePointerCapture(e.pointerId)
  })
  el.addEventListener('pointerleave', (e) => {
    console.log(88)
    e.stopImmediatePropagation()
    e.stopPropagation()
    e.preventDefault()
    el.releasePointerCapture(e.pointerId)
  })
  el.addEventListener('pointercancel', (e) => {
    console.log(77)
    e.stopImmediatePropagation()
    e.stopPropagation()
    e.preventDefault()
    el.releasePointerCapture(e.pointerId)
  })
  // el.addEventListener('pointerout', (e) => {
  //   console.log(44)
  //   el.releasePointerCapture(event.pointerId)
  // })

  // el.removeEventListener('pointerleave', onCancel)
  // el.addEventListener('pointerleave', onCancel)

  // el.removeEventListener('pointerleave', onLeave)
  // el.addEventListener('pointerleave', onLeave, { passive: true })

  // el.removeEventListener('pointerenter', onEnter)
  // el.addEventListener('pointerenter', onEnter, { passive: true })
}

function onEnter(event: PointerEvent) {
  console.log(444)
  const el = (event.target as HTMLElement).closest<HTMLElement>('button,summary')
  if (!el) return

  if (1 !== event.buttons) return

  if (el.closest('list-view').dataset.lastPointerCapture !== el.dataset.pointerCapture) return

  el.classList.add('active')

  // el.removeEventListener('pointerleave', onLeave)
  // el.addEventListener('pointerleave', onLeave, { passive: true, once: true })
}

function onLeave(event: PointerEvent) {
  console.log(333)
  const el = (event.target as HTMLElement).closest<HTMLElement>('button,summary')
  if (!el) return

  el.releasePointerCapture(event.pointerId)

  if (1 !== event.buttons) return

  if (el.closest('list-view').dataset.lastPointerCapture !== el.dataset.pointerCapture) return

  el.classList.remove('active')

  // el.removeEventListener('pointerenter', onEnter)
  // el.addEventListener('pointerenter', onEnter, { passive: true, once: true })
}

function onCancel(event: PointerEvent) {
  console.log(999, event.target)
  const el = (event.target as HTMLElement).closest<HTMLElement>('button,summary')
  if (!el) return

  el.releasePointerCapture(event.pointerId)

  delete el.dataset.pointerCapture

  self.requestAnimationFrame(() => {
    // el.removeEventListener('pointerleave', onCancel)
    el.classList.remove('active')
  })
}

const makePointerDownHandler = (target: HTMLElement) => {
    return onClick as EventListener
  },
  makePointerCancelHandler = (target: HTMLElement) => {
    return onCancel as EventListener
  }

export default function (t: HTMLElement) {
  const onListener = makePointerDownHandler(t),
    offListener = makePointerCancelHandler(t)

  return [
    { types: 'pointerdown', listener: onListener, addOptions: { passive: true } },
    { types: 'pointerup pointercancel', listener: offListener, addOptions: { passive: true } },
  ]
}
