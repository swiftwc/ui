export class CSSStyleObserver {
  #cleanups?: () => void

  constructor(target: HTMLElement) {}

  public async observe(target: Node, callback: (entry: MutationRecord) => void, options: MutationObserverInit) {
    this.#cleanups = onoff('transitionstart', this.#handleTransitionstart as EventListener, target).on()

    target.style.setProperty('transition', 'perspective-origin 1ms')
    target.style.setProperty('perspective-origin', 'top')
  }

  public async unobserve(target: Node) {
    this.#cleanups?.()
  }

  #handleTransitionstart = (evt: TransitionEvent) => {
